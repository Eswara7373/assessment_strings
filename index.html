        const countCharacters = (S) => 
        {
         let A = [];
         let countB = 0, countD = 0;
         for(let i = 0; i< S.length; i++){
           if(S.charAt(i) == "A")
           {
             countB = countB+1;
           }
           else if(S.charAt(i) =="D")
           {
             countD = countD+1;
           }
         }
         A.push(countB);
         A.push(countD);
         return(A);
        
        };

        var Palin_Check = (str) =>
        {let i =0  , j = str.length - 1 ;
        let palindrome = true ;
        while(i < j) {
           if(str[i] != str[j]) {
              palindrome = false ;
              break ;
           }
           i++ ; j-- ;
        }
        if(palindrome === true) {
           return "True";
        }
        else {
            return "False" ;
        }
        };

9.Reverse the String

        var Reverse_String = (str) => 
        {
          let reverse = "";
          let j = str.length - 1;
          for(let i=j;i>=0;i--){
            reverse = reverse+str[i];
          }
          return reverse;
        };
